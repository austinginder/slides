function mytheme_disable_gutenberg( $can_edit, $post_type ) {
  $disabled_post_types = array( "website", "customer" );
  if ( in_array($post_type, $disabled_post_types ) ) {
    return false;
  }
  return $can_edit;
}
add_filter( 'gutenberg_can_edit_post_type','mytheme_disable_gutenberg', 10, 2 );
